100 void Mixer::Tick() {
101  using cyclus::toolkit::RecordTimeSeries;
102  if (output.quantity() < output.capacity()) {
103  double tgt_qty = output.space();
104 
105  for (int i = 0; i < mixing_ratios.size(); i++) {
106  std::string name = "in_stream_" + std::to_string(i);
107  tgt_qty =
108  std::min(tgt_qty, streambufs[name].quantity() / mixing_ratios[i]);
109  }
110 
111  tgt_qty = std::min(tgt_qty, throughput);
112 
113  if (tgt_qty > 0) {
114  cyclus::Material::Ptr m;
115  for (int i = 0; i < mixing_ratios.size(); i++) {
116  std::string name = "in_stream_" + std::to_string(i);
117  double pop_qty = mixing_ratios[i] * tgt_qty;
118  if (i == 0) {
119  m = streambufs[name].Pop(pop_qty, cyclus::eps_rsrc());
120  } else {
121  cyclus::Material::Ptr m_ =
122  streambufs[name].Pop(pop_qty, cyclus::eps_rsrc());
123  m->Absorb(m_);
124  }
125  }
126  output.Push(m);
127  }
128  }
129  cyclus::toolkit::RecordTimeSeries<double>("supply"+out_commod, this, output.quantity());
130 }
